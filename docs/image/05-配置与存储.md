# 配置与存储

## 概述

聊天图片插件使用 SillyTavern 内置的 `extensionSettings` 系统存储所有数据，代码位于 `data.js` 中。**没有独立的 data.json 文件**，所有规则、规则集、角色集数据都嵌入到 `extensionSettings['chat-images']` 中统一管理。

---

## 1. 数据模块架构 (data.js)

### 模块标识

```javascript
export const MODULE_NAME = 'chat-images';
```

### 数据结构总览

```javascript
currentSettings = {
    // 插件设置
    enabled: true,              // 插件总开关
    lastActiveTab: 'charsets',  // 上次打开的标签页

    // 角色集列表
    charSets: [
        { id: 'char_xxx', name: '角色A', characterId: 'char_id' }
    ],

    // 规则集列表
    ruleSets: [
        { id: 'set_xxx', name: '表情组', charSetId: 'char_xxx' }
    ],

    // 规则列表
    rules: [
        {
            id: 'rule_xxx',
            name: '微笑检测',
            regex: '微笑|笑',
            enabled: true,
            order: 0,
            duration: 5,
            ruleSetId: 'set_xxx',
            _expanded: true,       // UI 折叠状态
            images: [
                { id: 'img_xxx', filename: 'smile.png', name: '微笑', weight: 50 }
            ]
        }
    ]
}
```

### 加载设置

```javascript
export function loadSettings() {
    const { extensionSettings } = SillyTavern.getContext();
    const saved = extensionSettings[MODULE_NAME];

    if (saved && saved.rules) {
        currentSettings = saved;
    } else {
        // 初始化默认值
        currentSettings = {
            enabled: true,
            lastActiveTab: 'charsets',
            charSets: [],
            ruleSets: [],
            rules: [],
        };
        extensionSettings[MODULE_NAME] = currentSettings;
        saveSettings();
    }
}
```

### 保存设置

```javascript
export function saveSettings() {
    const { extensionSettings, saveSettingsDebounced } = SillyTavern.getContext();
    extensionSettings[MODULE_NAME] = currentSettings;
    saveSettingsDebounced();
}
```

**关键点**：`saveSettingsDebounced()` 是 SillyTavern 提供的去抖保存函数，可避免频繁写入。

## 2. CRUD 操作

所有操作都在 `data.js` 中实现，直接修改 `currentSettings` 并调用 `saveSettings()`。

### 规则操作

```javascript
export function getRulesData() {
    return currentSettings;
}

export function getEnabledRules() {
    return currentSettings.rules.filter(r => r.enabled);
}

export function addRule(rule) {
    rule.id = generateId('rule');
    rule.order ??= 0;
    rule._expanded = true;
    currentSettings.rules.push(rule);
    saveSettings();
}

export function updateRule(ruleId, updates) {
    const idx = currentSettings.rules.findIndex(r => r.id === ruleId);
    if (idx !== -1) {
        Object.assign(currentSettings.rules[idx], updates);
        saveSettings();
    }
}

export function deleteRule(ruleId) {
    currentSettings.rules = currentSettings.rules.filter(r => r.id !== ruleId);
    saveSettings();
}
```

### 规则集操作

```javascript
export function getRuleSets() { return currentSettings.ruleSets; }
export function addRuleSet(rs) { rs.id = generateId('set'); currentSettings.ruleSets.push(rs); saveSettings(); }
export function updateRuleSet(id, updates) { /* find + assign + save */ }
export function deleteRuleSet(id) {
    currentSettings.ruleSets = currentSettings.ruleSets.filter(s => s.id !== id);
    // 同时将属于该规则集的规则的 ruleSetId 置空
    currentSettings.rules.forEach(r => { if (r.ruleSetId === id) r.ruleSetId = ''; });
    saveSettings();
}
```

### 角色集操作

```javascript
export function getCharSets() { return currentSettings.charSets; }
export function addCharSet(cs) { cs.id = generateId('char'); currentSettings.charSets.push(cs); saveSettings(); }
export function updateCharSet(id, updates) { /* find + assign + save */ }
export function deleteCharSet(id) {
    currentSettings.charSets = currentSettings.charSets.filter(c => c.id !== id);
    // 同时删除关联的规则集
    currentSettings.ruleSets = currentSettings.ruleSets.filter(s => s.charSetId !== id);
    saveSettings();
}
```

### 图片操作

```javascript
export function addImageToRule(ruleId, imageMeta) {

        const img = rule.images.find(i => i.id === imageId);

export function removeImageFromRule(ruleId, imageId) {
    const rule = currentSettings.rules.find(r => r.id === ruleId);
    if (rule) {
        rule.images = rule.images.filter(i => i.id !== imageId);
        saveSettings();
    }
}
```

## 4. 与旧版的区别

| 项目 | 旧版（文档中的） | 当前实现 |
|------|-----------------|---------|
| 数据文件 | 独立的 `data.json` | 统一在 `extensionSettings` 中 |
| 服务端端点 | `/api/plugins/chat-images/save-data` | 不需要自定义端点 |

| 图片上传 | `/api/plugins/chat-images/upload` | `/api/files/upload`（内置） |
| 图片删除 | 无 | `/api/files/delete`（内置） |

    return saveRulesData();

 * 向规则中添加图片

    rule.images.push(imageMeta);
    return saveRulesData();
}

/**
 * 更新图片权重

function updateImageWeight(ruleId, imageId, newWeight) {

    const img = rule.images.find(i => i.id === imageId);
    if (!img) return false;
    img.weight = newWeight;
    return saveRulesData();

/**

 */
function removeImageFromRule(ruleId, imageId) {
    const rule = getRuleById(ruleId);
    if (!rule) return false;
    rule.images = rule.images.filter(i => i.id !== imageId);

## 3. 数据迁移策略

当插件更新时，可能需要处理数据格式变更：

```javascript
/**
 * 数据迁移：确保旧版本数据能兼容新版本
 */
function migrateDataIfNeeded() {
    if (!rulesData.version || rulesData.version < 1) {
        // v0 → v1：初始迁移
        rulesData.version = 1;
        // 确保每条规则都有必要的字段
        for (const rule of rulesData.rules) {
            if (!rule.id) rule.id = `rule_${uuidv4().split('-')[0]}`;
            if (!rule.images) rule.images = [];
            if (rule.enabled === undefined) rule.enabled = true;
            for (const img of rule.images) {
                if (!img.id) img.id = `img_${uuidv4().split('-')[0]}`;
                if (!img.weight && img.weight !== 0) img.weight = 50;
            }
        }
        saveRulesData();
    }
    // 未来版本迁移：
    // if (rulesData.version < 2) { ... }
}
```

## 4. 与服务端通信

由于前端无法直接写入文件系统，需要通过 SillyTavern 的 API 进行文件操作。

### 注册服务端 API

在 `plugins.js` 或 `server.js` 中注册对应的路由：

```javascript
// 服务端伪代码
app.post('/api/plugins/chat-images/save-data', (req, res) => {
    const data = req.body;
    const filePath = path.join(__dirname, 'public/scripts/extensions/third-party/chat-images/data.json');
    fs.writeFileSync(filePath, JSON.stringify(data, null, 2));
    res.json({ success: true });
});

app.post('/api/plugins/chat-images/upload', (req, res) => {
    const { filename, data } = req.body;
    const base64Data = data.replace(/^data:image\/\w+;base64,/, '');
    const buffer = Buffer.from(base64Data, 'base64');
    const filePath = path.join(__dirname, 'public/scripts/extensions/third-party/chat-images/images', filename);
    fs.writeFileSync(filePath, buffer);
    res.json({ success: true, filename });
});
```

> **简化方案**: 如果不想写服务端代码，也可以考虑使用 `localStorage` 或 SillyTavern 的 `extensionSettings` 来存储规则数据（适用于规则较少的情况）。但独立的 `data.json` 方案更易于维护和备份。

## 5. 备选存储方案

### 使用 localStorage

```javascript
function loadRulesFromLocalStorage() {
    try {
        const data = localStorage.getItem('chat_images_rules');
        return data ? JSON.parse(data) : { version: 1, rules: [] };
    } catch {
        return { version: 1, rules: [] };
    }
}

function saveRulesToLocalStorage(data) {
    localStorage.setItem('chat_images_rules', JSON.stringify(data));
}
```

### 使用 SillyTavern 的 extensionSettings

```javascript
function loadRulesFromSettings() {
    const { extensionSettings } = SillyTavern.getContext();
    return extensionSettings[MODULE_NAME]?.rulesData || { version: 1, rules: [] };
}

function saveRulesToSettings(data) {
    const { extensionSettings, saveSettingsDebounced } = SillyTavern.getContext();
    extensionSettings[MODULE_NAME].rulesData = data;
    saveSettingsDebounced();
}
```

## 注意事项

1. **数据备份**: 在 `onDelete` 钩子中提示用户备份数据
2. **并发写入**: 使用防抖（debounce）避免频繁写入
3. **错误处理**: 所有读写操作都要有 try-catch 包裹
4. **版本兼容**: 始终使用 `version` 字段标记数据格式版本
5. **数据量控制**: 如果规则和图片很多，考虑分页加载
