# 配置与存储

## 概述

聊天图片插件需要持久化存储两类数据：

1. **插件设置** — 使用 SillyTavern 的 `extensionSettings` 系统存储
2. **规则与图片数据** — 存储正规则则和图片绑定关系，存储在 `data.json` 文件中

---

## 1. 插件设置（extensionSettings）

使用 SillyTavern 内置的 `extensionSettings` 对象存储插件的全局设置。

### 默认设置

```javascript
const MODULE_NAME = 'chat-images';

const defaultSettings = Object.freeze({
    enabled: true,                       // 插件总开关
    autoDetect: true,                    // AI 回复后自动检测
    showAsModal: true,                   // 匹配后是否弹窗展示
    insertToChat: false,                 // 匹配后是否插入到聊天
    modalAutoClose: true,                // 弹窗是否自动关闭
    modalAutoCloseDelay: 5000,           // 自动关闭延迟（毫秒）
    matchStrategy: 'first',              // 匹配策略: first, all, random
    notificationSound: false,            // 是否播放通知音
});
```

### 加载设置

```javascript
function loadSettings() {
    const { extensionSettings } = SillyTavern.getContext();

    if (!extensionSettings[MODULE_NAME]) {
        extensionSettings[MODULE_NAME] = structuredClone(defaultSettings);
    }

    // 确保所有默认键存在（兼容升级场景）
    for (const [key, value] of Object.entries(defaultSettings)) {
        if (extensionSettings[MODULE_NAME][key] === undefined) {
            extensionSettings[MODULE_NAME][key] = value;
        }
    }

    currentSettings = extensionSettings[MODULE_NAME];
}
```

### 保存设置

```javascript
function saveSettings() {
    const { extensionSettings, saveSettingsDebounced } = SillyTavern.getContext();
    extensionSettings[MODULE_NAME] = currentSettings;
    saveSettingsDebounced();
}
```

## 2. 规则与图片数据（data.json）

规则数据（正则与图片绑定）存储在独立的 `data.json` 文件中，便于直接读写和管理。

### 存储位置

```
third-party/chat-images/data.json
```

### 读写实现

```javascript
let rulesData = {
    version: 1,
    rules: [],
};

/**
 * 从 data.json 加载规则数据
 */
async function loadRulesData() {
    try {
        const response = await fetch('/scripts/extensions/third-party/chat-images/data.json');
        if (response.ok) {
            rulesData = await response.json();
        } else {
            // 文件不存在，使用默认空数据
            rulesData = { version: 1, rules: [] };
        }
    } catch (e) {
        console.warn('聊天图片插件: data.json 读取失败，使用默认数据');
        rulesData = { version: 1, rules: [] };
    }

    // 数据迁移：处理旧版本数据格式
    migrateDataIfNeeded();
}

/**
 * 保存规则数据到 data.json
 */
async function saveRulesData() {
    try {
        const response = await fetch('/api/plugins/chat-images/save-data', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(rulesData),
        });

        if (!response.ok) {
            throw new Error('保存失败');
        }
    } catch (e) {
        console.error('聊天图片插件: 保存 data.json 失败', e);
        toastr.error('规则数据保存失败');
    }
}
```

### 数据操作函数

```javascript
/**
 * 获取所有规则
 */
function getAllRules() {
    return rulesData.rules;
}

/**
 * 获取启用的规则
 */
function getEnabledRules() {
    return rulesData.rules.filter(r => r.enabled);
}

/**
 * 根据 ID 获取规则
 */
function getRuleById(ruleId) {
    return rulesData.rules.find(r => r.id === ruleId);
}

/**
 * 添加新规则
 */
function addRule(rule) {
    rule.id = rule.id || `rule_${uuidv4().split('-')[0]}`;
    rule.images = rule.images || [];
    rule.enabled = rule.enabled !== undefined ? rule.enabled : true;
    rulesData.rules.push(rule);
    return saveRulesData();
}

/**
 * 更新规则
 */
function updateRule(ruleId, updates) {
    const index = rulesData.rules.findIndex(r => r.id === ruleId);
    if (index === -1) return false;
    rulesData.rules[index] = { ...rulesData.rules[index], ...updates };
    return saveRulesData();
}

/**
 * 删除规则
 */
function deleteRule(ruleId) {
    rulesData.rules = rulesData.rules.filter(r => r.id !== ruleId);
    return saveRulesData();
}

/**
 * 向规则中添加图片
 */
function addImageToRule(ruleId, imageMeta) {
    const rule = getRuleById(ruleId);
    if (!rule) return false;
    rule.images.push(imageMeta);
    return saveRulesData();
}

/**
 * 更新图片权重
 */
function updateImageWeight(ruleId, imageId, newWeight) {
    const rule = getRuleById(ruleId);
    if (!rule) return false;
    const img = rule.images.find(i => i.id === imageId);
    if (!img) return false;
    img.weight = newWeight;
    return saveRulesData();
}

/**
 * 从规则中删除图片
 */
function removeImageFromRule(ruleId, imageId) {
    const rule = getRuleById(ruleId);
    if (!rule) return false;
    rule.images = rule.images.filter(i => i.id !== imageId);
    return saveRulesData();
}
```

## 3. 数据迁移策略

当插件更新时，可能需要处理数据格式变更：

```javascript
/**
 * 数据迁移：确保旧版本数据能兼容新版本
 */
function migrateDataIfNeeded() {
    if (!rulesData.version || rulesData.version < 1) {
        // v0 → v1：初始迁移
        rulesData.version = 1;
        // 确保每条规则都有必要的字段
        for (const rule of rulesData.rules) {
            if (!rule.id) rule.id = `rule_${uuidv4().split('-')[0]}`;
            if (!rule.images) rule.images = [];
            if (rule.enabled === undefined) rule.enabled = true;
            for (const img of rule.images) {
                if (!img.id) img.id = `img_${uuidv4().split('-')[0]}`;
                if (!img.weight && img.weight !== 0) img.weight = 50;
            }
        }
        saveRulesData();
    }
    // 未来版本迁移：
    // if (rulesData.version < 2) { ... }
}
```

## 4. 与服务端通信

由于前端无法直接写入文件系统，需要通过 SillyTavern 的 API 进行文件操作。

### 注册服务端 API

在 `plugins.js` 或 `server.js` 中注册对应的路由：

```javascript
// 服务端伪代码
app.post('/api/plugins/chat-images/save-data', (req, res) => {
    const data = req.body;
    const filePath = path.join(__dirname, 'public/scripts/extensions/third-party/chat-images/data.json');
    fs.writeFileSync(filePath, JSON.stringify(data, null, 2));
    res.json({ success: true });
});

app.post('/api/plugins/chat-images/upload', (req, res) => {
    const { filename, data } = req.body;
    const base64Data = data.replace(/^data:image\/\w+;base64,/, '');
    const buffer = Buffer.from(base64Data, 'base64');
    const filePath = path.join(__dirname, 'public/scripts/extensions/third-party/chat-images/images', filename);
    fs.writeFileSync(filePath, buffer);
    res.json({ success: true, filename });
});
```

> **简化方案**: 如果不想写服务端代码，也可以考虑使用 `localStorage` 或 SillyTavern 的 `extensionSettings` 来存储规则数据（适用于规则较少的情况）。但独立的 `data.json` 方案更易于维护和备份。

## 5. 备选存储方案

### 使用 localStorage

```javascript
function loadRulesFromLocalStorage() {
    try {
        const data = localStorage.getItem('chat_images_rules');
        return data ? JSON.parse(data) : { version: 1, rules: [] };
    } catch {
        return { version: 1, rules: [] };
    }
}

function saveRulesToLocalStorage(data) {
    localStorage.setItem('chat_images_rules', JSON.stringify(data));
}
```

### 使用 SillyTavern 的 extensionSettings

```javascript
function loadRulesFromSettings() {
    const { extensionSettings } = SillyTavern.getContext();
    return extensionSettings[MODULE_NAME]?.rulesData || { version: 1, rules: [] };
}

function saveRulesToSettings(data) {
    const { extensionSettings, saveSettingsDebounced } = SillyTavern.getContext();
    extensionSettings[MODULE_NAME].rulesData = data;
    saveSettingsDebounced();
}
```

## 注意事项

1. **数据备份**: 在 `onDelete` 钩子中提示用户备份数据
2. **并发写入**: 使用防抖（debounce）避免频繁写入
3. **错误处理**: 所有读写操作都要有 try-catch 包裹
4. **版本兼容**: 始终使用 `version` 字段标记数据格式版本
5. **数据量控制**: 如果规则和图片很多，考虑分页加载
