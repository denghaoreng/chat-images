## 8. 核心总结逻辑

### 8.1 触发条件判断

```javascript
async function getSummaryPromptForNow(context, force) {
    // 1. 间隔为 0 且非强制 → 跳过
    if (extension_settings.memory.promptInterval === 0 && !force) return '';

    // 2. 等待群组生成完成
    if (selected_group) await waitUntilCondition(() => is_group_generating === false, 1000, 10);

    // 3. 等待发送按钮释放（最多 30 秒）
    await waitUntilCondition(() => is_send_press === false, 30000, 100);

    // 4. 检查消息数量条件
    if (context.chat.length < extension_settings.memory.promptInterval && !force) return '';

    // 5. 计算上次总结以来的消息数和字数
    let messagesSinceLastSummary = 0;
    let wordsSinceLastSummary = 0;
    for (let i = context.chat.length - 1; i >= 0; i--) {
        if (context.chat[i].extra?.memory) break;
        messagesSinceLastSummary++;
        wordsSinceLastSummary += extractAllWords(context.chat[i].mes).length;
    }

    // 6. 检查是否满足触发条件
    let conditionSatisfied = false;
    if (messagesSinceLastSummary >= extension_settings.memory.promptInterval) conditionSatisfied = true;
    if (extension_settings.memory.promptForceWords && wordsSinceLastSummary >= extension_settings.memory.promptForceWords) conditionSatisfied = true;

    if (!conditionSatisfied && !force) return '';

    // 7. 返回替换了 {{words}} 宏的提示词
    return substituteParamsExtended(extension_settings.memory.prompt, { words: extension_settings.memory.promptWords });
}
```

### 8.4 构建原始总结 Prompt

```javascript
async function getRawSummaryPrompt(context, prompt) {
    const chat = context.chat.slice();
    const latestSummary = getLatestMemoryFromChat(chat);
    const latestSummaryIndex = getIndexOfLatestChatSummary(chat);
    chat.pop();  // 始终排除最后一条消息
    const chatBuffer = [];
    const PROMPT_SIZE = await getSourceContextSize();

    // 从上次总结之后的消息开始构建
    for (let index = latestSummaryIndex + 1; index < chat.length; index++) {
        const message = chat[index];
        if (message.is_system || !message.mes) continue;

        const entry = `${message.name}:\n${message.mes}`;
        chatBuffer.push(entry);

        // 检查 token 限制
        const tokens = await countSourceTokens(getMemoryString(true), 64);
        if (tokens > PROMPT_SIZE) { chatBuffer.pop(); break; }

        // 检查最大消息数限制
        if (maxMessagesPerRequest > 0 && chatBuffer.length >= maxMessagesPerRequest) break;
    }
    return { rawPrompt: getMemoryString(false), lastUsedIndex };
}
```

### 9.2 读取最新记忆

```javascript
function getLatestMemoryFromChat(chat) {
    if (!Array.isArray(chat) || !chat.length) return '';
    const reversedChat = chat.slice().reverse();
    reversedChat.shift();  // 排除最后一条
    for (let mes of reversedChat) {
        if (mes.extra && mes.extra.memory) return mes.extra.memory;
    }
    return '';
}
```

## 10. 上下文变更检测

```javascript
function isContextChanged(context) {
    const newContext = getContext();
    // 如果群组 ID、聊天 ID、角色 ID 任一发生变化，说明上下文已变
    if (newContext.groupId !== context.groupId
        || newContext.chatId !== context.chatId
        || (!newContext.groupId && (newContext.characterId !== context.characterId))) {
        console.log('Context changed, summary discarded');
        return true;
    }
    return false;
}
```

## 12. 总结回调（斜杠命令处理）

```javascript
async function summarizeCallback(args, text) {
    text = text.trim();
    if (!text) {
        return await forceSummarizeChat(isTrueBoolean(args.quiet));
    }

    const source = args.source || extension_settings.memory.source;
    const prompt = substituteParamsExtended((args.prompt || extension_settings.memory.prompt), { words: extension_settings.memory.promptWords });

    switch (source) {
        case 'extras': return await callExtrasSummarizeAPI(text);
        case 'main': return removeReasoningFromString(await generateRaw({
            prompt: text,
            systemPrompt: prompt,
            responseLength: extension_settings.memory.overrideResponseLength,
        }));

    }
}
```

## 14. 关键设计模式总结

| 模式 | 说明 | 代码位置 |
|------|------|----------|
| **模块命名空间** | 使用 `MODULE_NAME = '1_memory'` 避免命名冲突 | 顶部 |
| **默认设置** | 定义完整 `defaultSettings` 并用 `Object.freeze` 保护 | `loadSettings` |
| **防抖保存** | `debounce(() => getContext().saveChat(), timeout)` | 多处 |
| **API 调用锁** | `inApiCall` 标志防止并发总结 | 所有总结函数 |
| **上下文变更检测** | 在异步操作后检查上下文是否变化 | `isContextChanged` |
| **数据源切换 UI** | 使用 `data-summary-source` 属性控制显示 | `switchSourceControls` |
| **条件触发** | 基于消息数和字数双重条件触发总结 | `getSummaryPromptForNow` |
| **消息 Extra 存储** | 总结存储在 `mes.extra.memory` 中 | `setMemoryContext` |
| **降级处理** | UI 总结框优先，聊天记录作为后备 | `summaryMacroHandler` |
