# 最佳实践 — 安全、性能、兼容性

## 安全性

### ❌ 不要在 extensionSettings 中存储密钥

```javascript
// 错误 — 所有扩展都能访问 extensionSettings，且以明文存储
extensionSettings[MODULE_NAME].apiKey = 'secret_key_123';

// 注意：前端扩展没有安全存储密钥的方式。
// 如果需要处理敏感数据，使用服务端插件：
// https://docs.sillytavern.app/for-contributors/server-plugins/
```

### ✅ 用户输入净化

```javascript
// 验证输入类型
if (typeof userInput !== 'string') {
    toastr.error('无效的输入类型');
    return;
}

// 使用 DOMPurify 净化 HTML 输入
const { DOMPurify } = SillyTavern.libs;
const cleanInput = DOMPurify.sanitize(userInput);
```

### ❌ 避免使用 eval() 或 Function() 构造器

这些可以执行任意代码，存在安全风险。

---

### ✅ 使用事件委托处理动态元素

当列表项由 JS 动态生成时（如规则列表、图片列表），直接用 `$('.element').on('click', handler)` 会失效，因为元素还不存在。

```javascript
// ❌ 错误 — 动态生成的元素无法绑定
$('.rule-delete').on('click', function() { /* ... */ });

// ✅ 正确 — 使用事件委托到父容器
$('#rules-container').on('click', '.rule-delete', function() {
    const ruleId = $(this).data('rule-id');
    deleteRule(ruleId);
});
```

---

### ⚠️ 移动端适配要点

| 问题 | 解决方案 |
|------|---------|
| `100vh` 包含浏览器地址栏高度 | 使用 `100dvh`（dynamic viewport height）代替 |
| `dblclick` 在移动端无效 | 用点击时间戳检测（两次点击 < 400ms 视为双击） |
| 弹窗过宽 | 避免使用 `min-width` 固定值；不使用 `wide: true` 选项 |
| 触摸事件 | 确保按钮足够大（≥ 44px）便于点击 |

移动端双击检测示例：

```javascript
let lastClickTime = 0;

element.addEventListener('click', function() {
    const now = Date.now();
    if (now - lastClickTime < 400) {
        // 视为双击
        handleDoubleClick();
        lastClickTime = 0;
    } else {
        lastClickTime = now;
    }
});
```

## 性能

### ✅ 使用 localforage 存储大数据

```javascript
// 错误 — extensionSettings 会被频繁加载和保存
extensionSettings[MODULE_NAME].largeDataset = { /* 兆字节数据 */ };

// 正确 — 使用 localforage（IndexedDB/localStorage 的抽象）
const { localforage } = SillyTavern.libs;
await localforage.setItem(`${MODULE_NAME}_data`, largeData);

// 或使用 localStorage 存储小数据
localStorage.setItem(`${MODULE_NAME}_data`, JSON.stringify(smallData));
```

### ✅ 清理事件监听器，防止内存泄漏

```javascript
function cleanup() {
    eventSource.removeListener(event_types.MESSAGE_RECEIVED, handleMessage);
    document.getElementById('myElement')?.removeEventListener('click', handleClick);
}

// 在不需要时调用 cleanup()
```

### ✅ 不要阻塞 UI 线程

```javascript
// 异步处理 I/O 操作
async function processData() {
    const result = await fetch('/api/process');
    return result.json();
}

// 处理大量计算时，定期让出 UI 线程
async function heavyComputation(data) {
    for (let i = 0; i < data.length; i++) {
        // 处理数据块
        if (i % 1000 === 0) {
            await new Promise(resolve => setTimeout(resolve, 0)); // 让出 UI 线程
        }
    }
}
```

### ✅ 使用防抖减少保存频率

```javascript
import { debounce } from '../../utils.js';
import { debounce_timeout } from '../../constants.js';

const saveChatDebounced = debounce(() => getContext().saveChat(), debounce_timeout.relaxed);
```

## 兼容性

### ✅ 优先使用 getContext() 而非直接 import

```javascript
// ✅ 正确 — 稳定的 API
const { chat, characters, saveSettingsDebounced } = SillyTavern.getContext();

// ⚠️ 避免 — 内部结构可能随版本变化而改变
import { chat, characters } from '../../../../script.js';
```

### ✅ 使用唯一的模块名

```javascript
// ✅ 正确 — 具体且唯一
const MODULE_NAME = 'my_extension_name';

// ❌ 错误 — 太通用，可能与其他扩展冲突
const MODULE_NAME = 'settings';
```

### ✅ 正确初始化设置

```javascript
function loadSettings() {
    if (!extensionSettings[MODULE_NAME]) {
        extensionSettings[MODULE_NAME] = structuredClone(defaultSettings);
    }
    // 合并默认值，处理更新后的新键
    extensionSettings[MODULE_NAME] = SillyTavern.libs.lodash.merge(
        structuredClone(defaultSettings),
        extensionSettings[MODULE_NAME]
    );
}
```

### ✅ 检查和验证客户端版本

在 `manifest.json` 中设置 `minimum_client_version`：

```json
{
    "minimum_client_version": "1.12.0"
}
```

### ✅ 处理第三方插件的安装路径差异

两种安装方式路径不同：

```javascript
// 为所有用户安装: 路径相对于 /scripts/extensions/third-party/
// 为当前用户安装: 路径相对于 /scripts/extensions/

// 导入时使用相对路径需要考虑两种情况
```

### ✅ 模板渲染使用异步版本

```javascript
// ✅ 正确
const html = await renderExtensionTemplateAsync('my-ext', 'settings', data);

// ❌ 已废弃
const html = renderExtensionTemplate('my-ext', 'settings', data);
```

## 代码质量

### ✅ 使用 SillyTavern 内置库而非引入新依赖

在添加新的 npm 依赖前，检查 `SillyTavern.libs` 中是否已有可用的库。常见的内置库包括 `lodash`、`Fuse`、`DOMPurify`、`moment`、`yaml`、`Handlebars`、`Popper` 等。

### ✅ 为 TypeScript 用户添加类型声明

创建 `global.d.ts`：

```typescript
export {};

// 用户级安装
import '../../../../public/global';
// 服务端级安装
import '../../../../global';

// 额外类型声明
declare global {
    // 在这里添加全局类型声明
}
```

### ✅ 使用新版宏系统

```javascript
// ✅ 正确
macros.register('my_macro', {
    description: '我的宏',
    handler: () => { return 'value'; },
});

// ❌ 已废弃
registerMacro('my_macro', () => 'value');
```

### ✅ 使用新版斜杠命令系统

```javascript
// ✅ 正确
SlashCommandParser.addCommandObject(SlashCommand.fromProps({...}));

// ❌ 已废弃
registerSlashCommand('mycmd', callback, [], '帮助文本', 'group');
```

## 从零开始的完整插件模板

```javascript
// ========== index.js ==========
import { getContext } from '../../extensions.js';
import { eventSource, event_types } from '../../../script.js';

const MODULE_NAME = 'my_extension';

const defaultSettings = Object.freeze({
    enabled: true,
    someOption: 'default',
});

export async function init() {
    // 1. 加载设置
    loadSettings();

    // 2. 添加 UI
    const html = await renderExtensionTemplateAsync('third-party/my-extension', 'settings', {});
    $('#extensions_settings2').append(html);

    // 3. 绑定 UI 事件
    $('#my_option').on('input', onOptionInput);

    // 4. 注册事件
    eventSource.on(event_types.CHAT_CHANGED, onChatChanged);

    console.log(`[${MODULE_NAME}] 扩展已加载`);
}

function loadSettings() {
    if (!extension_settings[MODULE_NAME]) {
        extension_settings[MODULE_NAME] = structuredClone(defaultSettings);
    }
    for (const key of Object.keys(defaultSettings)) {
        if (extension_settings[MODULE_NAME][key] === undefined) {
            extension_settings[MODULE_NAME][key] = defaultSettings[key];
        }
    }
    // 同步到 UI
    $('#my_option').val(extension_settings[MODULE_NAME].someOption);
}

function onOptionInput() {
    extension_settings[MODULE_NAME].someOption = $(this).val();
    saveSettingsDebounced();
}

function onChatChanged() {
    console.log('聊天已切换');
}

// 可选：清理函数
export function onDisable() {
    eventSource.removeListener(event_types.CHAT_CHANGED, onChatChanged);
}
```
