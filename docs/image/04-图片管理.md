# 图片管理

## 概述

图片管理代码位于 `image.js` 中，负责图片的上传、删除、放大查看。图片通过 SillyTavern 内置的 `/api/files/upload` 上传到 `chat-images/` 目录，无需自定义服务端端点。

---

## 存储方案

### 目录结构

```
third-party/chat-images/
└── (图片存储在 SillyTavern 的 files/chat-images/ 目录下)
```

### 图片 URL 获取

```javascript
export function getImageUrl(filename) {
    return `/files/chat-images/${encodeURIComponent(filename)}`;
}
```

### 唯一文件名生成

```javascript
export function generateUniqueFilename(originalName) {
    const ext = originalName.split('.').pop();
    const timestamp = Date.now();
    const random = Math.random().toString(36).substring(2, 8);
    return `img_${timestamp}_${random}.${ext}`;
}
```

使用时间戳 + 随机字符串确保唯一性。

### 数据存储

图片元数据保存在 `extensionSettings['chat-images']` 中，嵌入到规则对象的 `images[]` 数组里：

```javascript
{
    "id": "img_abc123",
    "filename": "img_1718000000_a1b2c3.png",
    "name": "剑术攻击",
    "weight": 50
}
```

**注意**：`data.json` 已不再使用，所有数据统一存储在 `extensionSettings` 中。

## 上传流程 (image.js)

```
用户点击"添加图片"按钮
        │
        ▼
弹出文件选择对话框 (accept="image/*")
        │
        ▼
用户选择图片文件
        │
        ├──→ 验证文件类型（图片格式）
        │
        ├──→ 验证文件大小（建议 ≤ 5MB）
        │
        ▼
生成唯一文件名
        │
        ▼
通过 fetch → /api/files/upload 上传
        │
        ▼
创建图片元数据 → 添加到规则的 images[]
        │
        ▼
保存设置 → 更新 UI
```

### 单图片上传

```javascript
export async function handleImageUpload(ruleId) {
    const fileInput = document.createElement('input');
    fileInput.type = 'file';
    fileInput.accept = 'image/png,image/jpeg,image/gif,image/webp';

    fileInput.onchange = async function(e) {
        const file = e.target.files[0];
        if (!file) return;

        // 验证
        const allowedTypes = ['image/png', 'image/jpeg', 'image/gif', 'image/webp'];
        if (!allowedTypes.includes(file.type)) {
            toastr.warning('仅支持 PNG、JPEG、GIF、WebP 格式');
            return;
        }
        if (file.size > 5 * 1024 * 1024) {
            toastr.warning('图片大小不能超过 5MB');
            return;
        }

        const filename = generateUniqueFilename(file.name);
        const formData = new FormData();
        formData.append('file', file);
        formData.append('filename', filename);

        try {
            const response = await fetch('/api/files/upload', {
                method: 'POST',
                body: formData,
            });
            if (!response.ok) throw new Error('上传失败');

            // 添加到规则
            addImageToRule(ruleId, {
                id: generateId('img'),
                filename,
                name: file.name.replace(/\.[^/.]+$/, ''),
                weight: 50,
            });

            toastr.success('上传成功');
        } catch (err) {
            toastr.error('上传失败: ' + err.message);
        }
    };
    fileInput.click();
}
```

## 图片删除

```javascript
export async function deleteImageFile(filename) {
    const response = await fetch('/api/files/delete', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ path: `chat-images/${filename}` }),
    });
    if (!response.ok) throw new Error('删除失败');
}

// 在 rules-ui.js 中通过事件委托监听删除按钮
// 删除图片元数据 → 删除文件 → 更新 UI
```

## 图片放大查看

```javascript
export function chatImageEnlarge(imgElement) {
    // 创建全屏遮罩层
    const overlay = document.createElement('div');
    overlay.className = 'chat-images-enlarge-overlay';

    const enlarged = document.createElement('img');
    enlarged.className = 'chat-images-enlarged';
    enlarged.src = imgElement.src;

    overlay.appendChild(enlarged);
    document.body.appendChild(overlay);

    // 点击遮罩层关闭
    overlay.addEventListener('click', () => overlay.remove());

    // 图片自适应：width: 100vw, height: 100dvh
    // object-fit: contain, 保证图片完整显示
}
```

### 移动端适配

- 使用 `100dvh` 而非 `100vh`，避免移动端浏览器地址栏导致高度溢出
- 使用点击时间戳检测替代 `dblclick`（<400ms 判定为双击）
- `object-fit: contain` 确保图片完整显示

或者，可以使用更简单的方式——将图片数据以 Base64 形式存储在 data.json 中（适合小图片），但对于较大图片，建议直接写文件。

## 图片展示

### 缩略图列表

在规则编辑 UI 中，以缩略图网格展示绑定的图片：

```javascript
function renderRuleImages(rule) {
    const container = $(`#rule-${rule.id} .rule-images`);
    container.empty();

    for (const img of rule.images) {
        const imgUrl = getImageUrl(img.filename);

        const imgElement = $(`
            <div class="rule-image-item" data-image-id="${img.id}">
                <img class="rule-image-thumb" src="${imgUrl}" title="${img.name}">
                <div class="rule-image-weight">
                    <label>权重:</label>
                    <input type="range" min="0" max="100" value="${img.weight}"
                           class="image-weight-slider" data-image-id="${img.id}">
                    <span class="image-weight-value">${img.weight}</span>
                </div>
                <button class="rule-image-delete menu_button menu_button_icon"
                        data-image-id="${img.id}">
                    <i class="fa-solid fa-xmark"></i>
                </button>
            </div>
        `);

        container.append(imgElement);
    }
}

/**
 * 获取图片的访问 URL
 * @param {string} filename
 * @returns {string}
 */
function getImageUrl(filename) {
    return `/scripts/extensions/third-party/chat-images/images/${filename}`;
}
```

### 权重调整 UI

## 图片删除

```javascript
/**
 * 从规则中删除图片
 * @param {string} ruleId - 规则 ID
 * @param {string} imageId - 图片 ID
 */
async function deleteImage(ruleId, imageId) {
    const rules = getRules();
    const rule = rules.find(r => r.id === ruleId);
    if (!rule) return;

    // 从规则中移除图片
    const imageIndex = rule.images.findIndex(img => img.id === imageId);
    if (imageIndex === -1) return;

    const image = rule.images[imageIndex];
    rule.images.splice(imageIndex, 1);

    // 保存数据
    await saveRules(rules);

    // 重新渲染 UI
    renderRuleImages(rule);

    toastr.success(`图片 "${image.name}" 已删除`);
}
```

## 注意事项

1. **文件大小限制**: 建议限制单张图片不超过 5MB，避免加载过慢
2. **图片格式**: 建议支持 PNG、JPEG、GIF、WebP 四种常见格式
3. **文件名冲突**: 使用 UUID 生成唯一文件名，避免重名覆盖
4. **存储路径**: 确保 `images/` 目录在插件安装时自动创建
5. **缩略图缓存**: 可考虑生成缩略图（thumbnail）以加速列表渲染
6. **清理**: 在 `onDelete` 钩子中清理上传的图片文件
