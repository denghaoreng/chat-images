// ==================== 设置管理 ====================

function loadSettings() {
    const { extensionSettings } = getContext();

    if (!extensionSettings[MODULE_NAME]) {
        extensionSettings[MODULE_NAME] = structuredClone(defaultSettings);
    }

    for (const [key, value] of Object.entries(defaultSettings)) {
        if (extensionSettings[MODULE_NAME][key] === undefined) {
            extensionSettings[MODULE_NAME][key] = value;
        }
    }

    currentSettings = extensionSettings[MODULE_NAME];
}

function saveSettings() {
    const { extensionSettings, saveSettingsDebounced } = getContext();
    extensionSettings[MODULE_NAME] = currentSettings;
    saveSettingsDebounced();
}

// ==================== 数据管理 ====================

async function loadRulesData() {
    try {
        const response = await fetch('/scripts/extensions/third-party/chat-images/data.json');
        if (response.ok) {
            rulesData = await response.json();
        }
    } catch (e) {
        console.warn('聊天图片插件: data.json 读取失败，使用默认数据');
        rulesData = { version: 1, rules: [] };
    }
    migrateDataIfNeeded();
}

async function saveRulesData() {
    try {
        const response = await fetch('/api/plugins/chat-images/save-data', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(rulesData),
        });
        if (!response.ok) throw new Error('保存失败');
    } catch (e) {
        console.error('聊天图片插件: 保存 data.json 失败', e);
        toastr.error('规则数据保存失败');
    }
}

function migrateDataIfNeeded() {
    if (!rulesData.version || rulesData.version < 1) {
        rulesData.version = 1;
        for (const rule of rulesData.rules) {
            if (!rule.id) rule.id = `rule_${uuidv4().split('-')[0]}`;
            if (!rule.images) rule.images = [];
            if (rule.enabled === undefined) rule.enabled = true;
            for (const img of rule.images) {
                if (!img.id) img.id = `img_${uuidv4().split('-')[0]}`;
                if (!img.weight && img.weight !== 0) img.weight = 50;
            }
        }
        saveRulesData();
    }
}

// ==================== 规则 CRUD ====================

function getEnabledRules() {
    return rulesData.rules.filter(r => r.enabled);
}

function addRule(rule) {
    const newRule = {
        id: `rule_${uuidv4().split('-')[0]}`,
        name: rule.name || '新规则',
        regex: rule.regex || '',
        enabled: true,
        images: [],
    };
    rulesData.rules.push(newRule);
    saveRulesData();
    return newRule;
}

function deleteRule(ruleId) {
    rulesData.rules = rulesData.rules.filter(r => r.id !== ruleId);
    saveRulesData();
}

function updateRule(ruleId, updates) {
    const rule = rulesData.rules.find(r => r.id === ruleId);
    if (!rule) return;
    Object.assign(rule, updates);
    saveRulesData();
}

// ==================== 图片管理 ====================

function getImageUrl(filename) {
    return `/scripts/extensions/third-party/chat-images/images/${filename}`;
}

function generateUniqueFilename(originalName) {
    const ext = originalName.split('.').pop();
    return `img_${uuidv4().split('-')[0]}.${ext}`;
}

async function handleImageUpload(ruleId) {
    const fileInput = document.createElement('input');
    fileInput.type = 'file';
    fileInput.accept = 'image/png,image/jpeg,image/gif,image/webp';

    fileInput.onchange = async function(e) {
        const file = e.target.files[0];
        if (!file) return;

        const allowedTypes = ['image/png', 'image/jpeg', 'image/gif', 'image/webp'];
        if (!allowedTypes.includes(file.type)) {
            toastr.warning('仅支持 PNG、JPEG、GIF、WebP 格式的图片');
            return;
        }

        const maxSize = 5 * 1024 * 1024;
        if (file.size > maxSize) {
            toastr.warning('图片大小不能超过 5MB');
            return;
        }

        const filename = generateUniqueFilename(file.name);

        const reader = new FileReader();
        reader.onload = async function(ev) {
            // 保存图片文件
            await saveImageFile(filename, ev.target.result);

            // 创建元数据
            const imageMeta = {
                id: `img_${uuidv4().split('-')[0]}`,
                filename: filename,
                originalName: file.name,
                name: file.name.replace(/\.[^/.]+$/, ''),
                weight: 50,
                uploadDate: new Date().toISOString(),
                fileSize: file.size,
            };

            // 添加到规则
            const rule = rulesData.rules.find(r => r.id === ruleId);
            if (rule) {
                rule.images.push(imageMeta);
                await saveRulesData();
                renderRuleList();
                toastr.success('图片上传成功');
            }
        };
        reader.readAsDataURL(file);
    };

    fileInput.click();
}

async function saveImageFile(filename, fileData) {
    const response = await fetch('/api/plugins/chat-images/upload', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ filename, data: fileData }),
    });
    if (!response.ok) throw new Error('图片保存失败');
}

// ==================== 加权随机选择 ====================

function selectImageByWeight(images) {
    if (!images || images.length === 0) return null;

    const totalWeight = images.reduce((sum, img) => sum + (img.weight || 1), 0);
    if (totalWeight <= 0) {
        return images[Math.floor(Math.random() * images.length)];
    }

    let random = Math.random() * totalWeight;
    for (const img of images) {
        random -= (img.weight || 1);
        if (random <= 0) return img;
    }
    return images[images.length - 1];
}

function showImageModal(image) {
    const modal = document.getElementById('chat-images-modal');
    const modalImg = document.getElementById('chat-images-modal-img');

// ==================== 事件监听 ====================

function registerEventListeners() {
    const { eventSource, event_types } = getContext();

    // 先移除旧监听，防止重复
    eventSource.removeListener(event_types.MESSAGE_RECEIVED, onMessageReceived);
    eventSource.removeListener(event_types.MESSAGE_SWIPED, onMessageSwiped);

    // 注册新监听
    eventSource.on(event_types.MESSAGE_RECEIVED, onMessageReceived);
    eventSource.on(event_types.MESSAGE_SWIPED, onMessageSwiped);
}

function onMessageReceived(data) {
    if (!currentSettings.enabled || !currentSettings.autoDetect) return;
    if (data.is_user || data.is_system) return;
    if (!data.mes) return;

    performMatch(data.mes);
}

function onMessageSwiped() {
    if (!currentSettings.enabled || !currentSettings.autoDetect) return;

    const { chat } = getContext();
    const currentMessage = chat[chat.length - 1];
    if (!currentMessage || currentMessage.is_user) return;

    performMatch(currentMessage.mes);
}

// ==================== 匹配逻辑 ====================

function performMatch(text) {
    if (!text) return;

    const enabledRules = getEnabledRules();
    if (enabledRules.length === 0) return;

    const matchedImages = [];

    for (const rule of enabledRules) {
        try {
            const regex = new RegExp(rule.regex, 'gi');
            if (regex.test(text)) {
                const selectedImage = selectImageByWeight(rule.images);
                if (selectedImage) {
                    matchedImages.push(selectedImage);
                    if (currentSettings.matchStrategy === 'first') break;
                }
            }
        } catch (e) {
            console.error(`聊天图片插件: 规则 "${rule.name}" 的正则执行错误`, e);
        }
    }

    if (matchedImages.length === 0) return;

    // 根据策略决定展示哪些图片
    let imagesToShow = [];
    switch (currentSettings.matchStrategy) {
        case 'first':
            imagesToShow = [matchedImages[0]];
            break;
        case 'all':
            imagesToShow = matchedImages;
            break;
        case 'random':
            imagesToShow = [matchedImages[Math.floor(Math.random() * matchedImages.length)]];
            break;
        default:
            imagesToShow = [matchedImages[0]];
    }

    // 展示图片
    for (const image of imagesToShow) {
        if (currentSettings.showAsModal) {
            showImageModal(image);
        }
        if (currentSettings.insertToChat) {
            insertImageToChat(getImageUrl(image.filename));
        }
    }
}

// ==================== UI 渲染 ====================

function renderRuleList() {
    const container = $('#chat-images-rule-list');
    container.empty();

    if (rulesData.rules.length === 0) {
        container.html('<div class="text-center" style="padding:20px;opacity:0.5;">暂无规则，点击下方按钮添加</div>');
        return;
    }

    for (const rule of rulesData.rules) {
        const ruleElement = $(`
            <div class="rule-item" data-rule-id="${rule.id}">
                <div class="flex-container alignitemscenter margin0">
                    <input class="rule-name text_pole flex1" type="text" value="${escapeHtml(rule.name)}" placeholder="规则名称">
                    <label class="checkbox_label marginLeft5">
                        <input type="checkbox" class="rule-enabled" ${rule.enabled ? 'checked' : ''}>
                        <span>启用</span>
                    </label>
                    <button class="rule-delete menu_button menu_button_icon marginLeft5" title="删除规则">
                        <i class="fa-solid fa-trash-can"></i>
                    </button>
                </div>
                <div class="marginTop5">
                    <input class="rule-regex text_pole wide100p" type="text"
                           value="${escapeHtml(rule.regex)}" placeholder="输入正则表达式，如: 攻击|战斗|魔法">
                </div>
                <div class="rule-images marginTop5 flex-container flexWrap gap5px">
                    ${renderRuleImages(rule)}
                </div>
                <div class="flex-container marginTop5">
                    <button class="rule-add-image menu_button" data-rule-id="${rule.id}">
                        <i class="fa-solid fa-upload"></i> 添加图片
                    </button>
                </div>
            </div>
        `);

        container.append(ruleElement);
        bindRuleEvents(ruleElement, rule);
    }
}

function renderRuleImages(rule) {
    if (rule.images.length === 0) {
        return '<span style="opacity:0.5;font-size:0.9em;">暂无图片</span>';
    }

    return rule.images.map(img => `
        <div class="rule-image-item" data-image-id="${img.id}">
            <img class="rule-image-thumb" src="${getImageUrl(img.filename)}"
                 title="${escapeHtml(img.name)}" loading="lazy">
            <div class="rule-image-weight">
                <input type="range" min="0" max="100" value="${img.weight}"
                       class="image-weight-slider" data-image-id="${img.id}">
                <span class="image-weight-value">${img.weight}</span>
            </div>
            <button class="rule-image-delete menu_button menu_button_icon" title="删除图片">
                <i class="fa-solid fa-xmark"></i>
            </button>
        </div>
    `).join('');
}

function bindRuleEvents(ruleElement, rule) {
    // 规则名称变更
    ruleElement.find('.rule-name').on('input', function() {
        updateRule(rule.id, { name: $(this).val() });
    });

    // 正则变更
    ruleElement.find('.rule-regex').on('input', function() {
        updateRule(rule.id, { regex: $(this).val() });
    });

    // 启用/禁用
    ruleElement.find('.rule-enabled').on('change', function() {
        updateRule(rule.id, { enabled: $(this).is(':checked') });
    });

    // 删除规则
    ruleElement.find('.rule-delete').on('click', function() {
        if (confirm(`确定删除规则 "${rule.name}" 吗？`)) {
            deleteRule(rule.id);
            renderRuleList();
        }
    });

    // 添加图片
    ruleElement.find('.rule-add-image').on('click', function() {
        handleImageUpload(rule.id);
    });

    // 图片权重变更（委托）
    ruleElement.find('.image-weight-slider').on('input', function() {
        const imageId = $(this).data('image-id');
        const value = parseInt($(this).val());
        $(this).siblings('.image-weight-value').text(value);

        const img = rule.images.find(i => i.id === imageId);
        if (img) {
            img.weight = value;
            saveRulesData();
        }
    });

    // 删除图片
    ruleElement.find('.rule-image-delete').on('click', function() {
        const imageId = $(this).closest('.rule-image-item').data('image-id');
        rule.images = rule.images.filter(i => i.id !== imageId);
        saveRulesData();
        renderRuleList();
    });
}

// ==================== 工具函数 ====================

function escapeHtml(text) {
    if (!text) return '';
    return text.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;').replace(/'/g, '&#039;');
}

// ==================== UI 事件绑定 ====================

function bindUIEvents() {
    // 添加规则按钮
    $(document).off('click', '#chat-images-add-rule').on('click', '#chat-images-add-rule', function() {
        addRule({ name: '新规则', regex: '' });
        renderRuleList();
    });

    // 设置面板控件绑定
    $(document).off('change', '#chat-images-enabled').on('change', '#chat-images-enabled', function() {
        currentSettings.enabled = $(this).is(':checked');
        saveSettings();
    });

    $(document).off('change', '#chat-images-auto-detect').on('change', '#chat-images-auto-detect', function() {
        currentSettings.autoDetect = $(this).is(':checked');
        saveSettings();
    });

    $(document).off('change', '#chat-images-show-modal').on('change', '#chat-images-show-modal', function() {
        currentSettings.showAsModal = $(this).is(':checked');
        saveSettings();
    });

    $(document).off('change', '#chat-images-insert-chat').on('change', '#chat-images-insert-chat', function() {
        currentSettings.insertToChat = $(this).is(':checked');
        saveSettings();
    });

    $(document).off('change', '#chat-images-auto-close').on('change', '#chat-images-auto-close', function() {
        currentSettings.modalAutoClose = $(this).is(':checked');
        saveSettings();
    });

    $(document).off('change', '#chat-images-close-delay').on('change', '#chat-images-close-delay', function() {
        currentSettings.modalAutoCloseDelay = parseInt($(this).val()) * 1000;
        saveSettings();
    });

    $(document).off('change', '#chat-images-match-strategy').on('change', '#chat-images-match-strategy', function() {
        currentSettings.matchStrategy = $(this).val();
        saveSettings();
    });
}

## 文件清单

创建插件时，需要准备以下文件：

```
third-party/chat-images/
├── manifest.json      # 插件清单
├── index.js           # 主逻辑（完整源码如上）
├── style.css          # 样式文件
├── settings.html      # 设置面板模板
├── images/            # 图片存储目录（会自动创建）
├── data.json          # 规则数据（初始内容: {"version":1,"rules":[]}）
└── docs/              # 文档目录
    └── ...
```

---

## 初始 data.json

```json
{
    "version": 1,
    "rules": []
}
```

## 部署步骤

1. 在 `public/scripts/extensions/third-party/` 下创建 `chat-images/` 目录
2. 将以上所有文件放入该目录
3. 创建 `images/` 空目录
4. 创建 `data.json` 写入初始数据 `{"version":1,"rules":[]}`
5. 重启 SillyTavern 服务端
6. 在扩展管理中找到"聊天图片"并启用
7. 点击顶栏的图片图标开始配置规则
