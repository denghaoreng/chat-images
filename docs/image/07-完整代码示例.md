async function saveImageFile(filename, fileData) {
    const response = await fetch('/api/plugins/chat-images/upload', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ filename, data: fileData }),
    });
    if (!response.ok) throw new Error('图片保存失败');
}

// ==================== 加权随机选择 ====================

function selectImageByWeight(images) {
    if (!images || images.length === 0) return null;

    const totalWeight = images.reduce((sum, img) => sum + (img.weight || 1), 0);
    if (totalWeight <= 0) {
        return images[Math.floor(Math.random() * images.length)];
    }

    let random = Math.random() * totalWeight;
    for (const img of images) {
        random -= (img.weight || 1);
        if (random <= 0) return img;
    }
    return images[images.length - 1];
}

// ==================== 事件监听 ====================

function registerEventListeners() {
    const { eventSource, event_types } = getContext();

    // 先移除旧监听，防止重复
    eventSource.removeListener(event_types.MESSAGE_RECEIVED, onMessageReceived);
    eventSource.removeListener(event_types.MESSAGE_SWIPED, onMessageSwiped);

    // 注册新监听
    eventSource.on(event_types.MESSAGE_RECEIVED, onMessageReceived);
    eventSource.on(event_types.MESSAGE_SWIPED, onMessageSwiped);
}

function onMessageReceived(data) {
    if (!currentSettings.enabled || !currentSettings.autoDetect) return;
    if (data.is_user || data.is_system) return;
    if (!data.mes) return;

    performMatch(data.mes);
}

function onMessageSwiped() {
    if (!currentSettings.enabled || !currentSettings.autoDetect) return;

    const { chat } = getContext();
    const currentMessage = chat[chat.length - 1];
    if (!currentMessage || currentMessage.is_user) return;

    performMatch(currentMessage.mes);
}

// ==================== 匹配逻辑 ====================

function performMatch(text) {
    if (!text) return;

    const enabledRules = getEnabledRules();
    if (enabledRules.length === 0) return;

    const matchedImages = [];

    for (const rule of enabledRules) {
        try {
            const regex = new RegExp(rule.regex, 'gi');
            if (regex.test(text)) {
                const selectedImage = selectImageByWeight(rule.images);
                if (selectedImage) {
                    matchedImages.push(selectedImage);
                    if (currentSettings.matchStrategy === 'first') break;
                }
            }
        } catch (e) {
            console.error(`聊天图片插件: 规则 "${rule.name}" 的正则执行错误`, e);
        }
    }

    if (matchedImages.length === 0) return;

    // 根据策略决定展示哪些图片
    let imagesToShow = [];
    switch (currentSettings.matchStrategy) {
        case 'first':
            imagesToShow = [matchedImages[0]];
            break;
        case 'all':
            imagesToShow = matchedImages;
            break;
        case 'random':
            imagesToShow = [matchedImages[Math.floor(Math.random() * matchedImages.length)]];
            break;
        default:
            imagesToShow = [matchedImages[0]];
    }

    // 展示图片
    for (const image of imagesToShow) {
        if (currentSettings.showAsModal) {
            showImageModal(image);
        }
        if (currentSettings.insertToChat) {
            insertImageToChat(getImageUrl(image.filename));
        }
    }
}

function renderRuleImages(rule) {
    if (rule.images.length === 0) {
        return '<span style="opacity:0.5;font-size:0.9em;">暂无图片</span>';
    }

    return rule.images.map(img => `
        <div class="rule-image-item" data-image-id="${img.id}">
            <img class="rule-image-thumb" src="${getImageUrl(img.filename)}"
                 title="${escapeHtml(img.name)}" loading="lazy">
            <div class="rule-image-weight">
                <input type="range" min="0" max="100" value="${img.weight}"
                       class="image-weight-slider" data-image-id="${img.id}">
                <span class="image-weight-value">${img.weight}</span>
            </div>
            <button class="rule-image-delete menu_button menu_button_icon" title="删除图片">
                <i class="fa-solid fa-xmark"></i>
            </button>
        </div>
    `).join('');
}

function bindRuleEvents(ruleElement, rule) {
    // 规则名称变更
    ruleElement.find('.rule-name').on('input', function() {
        updateRule(rule.id, { name: $(this).val() });
    });

    // 正则变更
    ruleElement.find('.rule-regex').on('input', function() {
        updateRule(rule.id, { regex: $(this).val() });
    });

    // 启用/禁用
    ruleElement.find('.rule-enabled').on('change', function() {
        updateRule(rule.id, { enabled: $(this).is(':checked') });
    });

    // 删除规则
    ruleElement.find('.rule-delete').on('click', function() {
        if (confirm(`确定删除规则 "${rule.name}" 吗？`)) {
            deleteRule(rule.id);
            renderRuleList();
        }
    });

    // 添加图片
    ruleElement.find('.rule-add-image').on('click', function() {
        handleImageUpload(rule.id);
    });

    // 图片权重变更（委托）
    ruleElement.find('.image-weight-slider').on('input', function() {
        const imageId = $(this).data('image-id');
        const value = parseInt($(this).val());
        $(this).siblings('.image-weight-value').text(value);

        const img = rule.images.find(i => i.id === imageId);
        if (img) {
            img.weight = value;
            saveRulesData();
        }
    });

    // 删除图片
    ruleElement.find('.rule-image-delete').on('click', function() {
        const imageId = $(this).closest('.rule-image-item').data('image-id');
        rule.images = rule.images.filter(i => i.id !== imageId);
        saveRulesData();
        renderRuleList();
    });
}

function escapeHtml(text) {
    if (!text) return '';
    return text.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;').replace(/'/g, '&#039;');
}

## 部署步骤

1. 在 `public/scripts/extensions/third-party/` 下创建 `chat-images/` 目录
2. 将以上所有文件放入该目录
3. 创建 `images/` 空目录
4. 创建 `data.json` 写入初始数据 `{"version":1,"rules":[]}`
5. 重启 SillyTavern 服务端
6. 在扩展管理中找到"聊天图片"并启用
7. 点击顶栏的图片图标开始配置规则
